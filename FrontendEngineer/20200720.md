```
//数组反转
function Reverse(arr = []) {
    return arr.reduceRight((t, v) => (t.push(v), t), []);//t.push返回数组长度   (t.push(v), t)可以用(t.push(v)&& t)
}
Reverse([1, 2, 3, 4, 5]); // [5, 4, 3, 2, 1]
//去重
function Uniq(arr=[]){
  return arr.reduce((t,v)=>{
    console.log(t,v)
    return t.includes(v)?t:[...t,v]},[])
}
```
